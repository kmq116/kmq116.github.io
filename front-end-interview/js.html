    # 数组去重

    indexOf 循环遍历

    function unique(arr) {
    const result = []
    arr.forEach(function(item) {
        if (result.indexOf(item) == -1) {
            result.push(item)
        }
    })
    
    return result
}

    filter

    var array = [1, 2, 1, 1, '1'];

function unique(array) {
    var res = array.filter(function(item, index, array){
        return array.indexOf(item) === index;
    })
    return res;
}

console.log(unique(array));
